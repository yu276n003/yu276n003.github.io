    <script>
        let isDrawing = false;

        function startDraw() {
            if (isDrawing) return;

            const inputEl = document.getElementById('optionsInput');
            // 讀取目前的選項並過濾空行
            let options = inputEl.value.split('\n').map(opt => opt.trim()).filter(opt => opt !== "");

            if (options.length === 0) {
                alert("名單已經空了！請重新輸入選項。");
                return;
            }

            isDrawing = true;
            const resultDisplay = document.getElementById('resultDisplay');
            let counter = 0;
            let finalSelectedIndex = 0;
            
            // 跑馬燈跳動效果
            const interval = setInterval(() => {
                const randomIndex = Math.floor(Math.random() * options.length);
                resultDisplay.innerText = options[randomIndex];
                counter++;
                
                if (counter > 15) {
                    clearInterval(interval);
                    
                    // 決定最終中獎者
                    finalSelectedIndex = Math.floor(Math.random() * options.length);
                    const winner = options[finalSelectedIndex];
                    
                    resultDisplay.innerText = "🎉 " + winner;

                    // 觸發全螢幕彩帶特效
                    confetti({
                        particleCount: 100,
                        spread: 70,
                        origin: { y: 0.6 }
                    });

                    // 如果開啟淘汰賽模式，將該選項移除
                    if (document.getElementById('deleteMode').checked) {
                        options.splice(finalSelectedIndex, 1);
                        inputEl.value = options.join('\n'); // 更新文字框內容
                    }

                    isDrawing = false;
                }
            }, 100);
        }

        // 動態切換網頁配色主題
        function changeTheme(name, primary, hover, gradient) {
            document.documentElement.style.setProperty('--bg-gradient', gradient);
            document.documentElement.style.setProperty('--primary-color', primary);
            document.documentElement.style.setProperty('--primary-hover', hover);
        }
    </script>
