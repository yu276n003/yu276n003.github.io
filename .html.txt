    <script>
        function draw() 
            let list = document.getElementById('inputs').value.split('\n').filter(x => x.trim() !== "");
            
            if (list.length === 0) {
                alert("請輸入選項！");
                return;
            }
            
            // 隨機選一個
            let randomIndex = Math.floor(Math.random() * list.length);
            document.getElementById('result').innerText = "🎉 " + list[randomIndex];
        }
    </script>
